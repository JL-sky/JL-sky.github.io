#include<arpa/inet.h>
#include<netinet/in.h>
using namespace std;
int main()
{
        // #include <sys/types.h>          /* See NOTES */
    // #include <sys/socket.h>
    //返回文件描述符
    //domain:协议族；type：socket类型：protocol：协议，为0表示自动创建于type对应的协议
    // int socket(int domain, int type, int protocol);

    //创建socket
        int cfd=socket(AF_INET,SOCK_STREAM,0);//AF_INET：ipV4协议；SOCK_STREAM：tcp协议
    
        if(cfd<0)
        {
                perror("socket error");
                return -1;
        }
    //链接服务端
        struct sockaddr_in serv;
        serv.sin_family=AF_INET;
        serv.sin_port=htons(8888);
        inet_pton(AF_INET,"127.0.0.1",&serv.sin_addr.s_addr);
        int ret=connect(cfd,(struct sockaddr *)&serv,sizeof(serv));
        if(ret<0)
        {
                perror("connect error");
                return -1;
        }
    //想服务器发送数据并接受来自服务器的数据
    
        int n=0;
        char buf[256];
        while(1)
        {
                //读标准输入数据
                memset(buf,0,sizeof(buf));
                n=read(STDIN_FILENO,buf,sizeof(buf));
                //发送数据
                write(cfd,buf,n);
                //读取服务器数据
                memset(buf,0,sizeof(buf));
                n=read(cfd,buf,sizeof(buf));
                if(n<=0)
                {
                        cout<<"read error or connect closed!"<<endl;
                        break;
                }
    
        }
        close(cfd);
        return 1;

}

多进程并发服务器

#include<stdlib.h>
#include<sys/types.h>
#include<sys/socket.h>
#include<unistd.h>
#include<arpa/inet.h>
#include<netinet/in.h>
#include<netdb.h>
#include<ctype.h>
#include<stdio.h>
#include<string.h>
int main()
{
        //创建socket
        int lfd=socket(AF_INET,SOCK_STREAM,0);

        //将端口号与if绑定到socket
        struct sockaddr_in serv;
        memset(&serv,0,sizeof(serv));
        serv.sin_family=AF_INET;
        serv.sin_port=htons(8888);
        serv.sin_addr.s_addr=htonl(INADDR_ANY);
    
        bind(lfd,(struct sockaddr*)&serv,sizeof(serv));
    
        //监听
        listen(lfd,128);
    
        //收发数据
    
        pid_t pid;
        int cfd;
        int socklen=sizeof(struct sockaddr_in);
        struct sockaddr_in clientaddr;
        while(1)
        {
            //从等待队列中取出一个客户端
            cfd=accept(lfd,(struct sockaddr*)&clientaddr,(socklen_t*)&socklen);
            printf("客户端%s已经链接\n",inet_ntoa(clientaddr.sin_addr));
            // 创建一个新的进程，该进程从父进程复制而来，只有少量数据与父进程不同
            // fork（）函数会返回两个返回值pid，
            // 若pid==0，表示当前运行进程为子进程，若pid>0,表示当前运行进程为父进程，且pid为子进程的进程号
            pid=fork();
            if(pid<0)//子进程创建失败
            {
                perror("fork error");
                exit(-1);
            }
            else if(pid>0)//父进程
            {
                close(cfd);//让父进程只进行监听
            }
            else if(pid==0)//子进程
            {
                close(lfd);//关闭监听描述符，让子进程只收发数据
                int n;
                char buff[1024];
                while(1)
                {
                    n=read(cfd,buff,sizeof(buff));
                    printf("端口号为[%d]",ntohs(clientaddr.sin_port));
                    if(n<=0)
                    {
                        printf("read error or client closed,n==%d\n",n);
                        break;
                    }
                    printf("n=%d,buff=%s\n",n,buff);
                    write(cfd,"OK\n",n);
                }
                close(cfd);//对应子进程收发完数据后关闭文件描述符
                exit(0);//防止子进程继续创建子进程
            }
        }
        close(lfd);//父进程关闭监听描述符
        return 1;

}

多线程并发服务器

#include<iostream>
#include<string.h>
#include <strings.h>
#include <sys/socket.h>
#include<sys/types.h>
#include<unistd.h>
#include<arpa/inet.h>
#include<netinet/in.h>
#include<pthread.h>
using namespace std;
void* mythread(void* arg)//子线程执行函数
{
        int cfd=*(int *)arg;
        int n;
        char buff[1024];
        while(1)
        {
                memset(buff,0,sizeof(buff));
                n=read(cfd,buff,sizeof(buff));
                if(n<=0)
                {
                        cout<<"read error or client closed!"<<endl;
                        break;
                }

                cout<<"共接受"<<n<<"个字符："<<buff<<endl;
                //发送数据
                write(cfd,buff,n);
        }
        close(cfd);
        pthread_exit(NULL);

}
int main()
{
        //创建socket
        int lfd=socket(AF_INET, SOCK_STREAM, 0);

        struct sockaddr_in serv;
        bzero(&serv, sizeof(serv));
        serv.sin_family=AF_INET;
        serv.sin_port=htons(8888);//主机字节序转换为网络字节序
        serv.sin_addr.s_addr=htonl(INADDR_ANY);
        //绑定
        bind(lfd,(struct sockaddr*)&serv,sizeof(serv));
        //监听
        listen(lfd,128);
    
        int cfd;
        pthread_t tid;
        while(1)
        {
                //接受客户端
                cfd=accept(lfd,NULL,NULL);
                //创建子线程
                cfd=pthread_create(&tid,NULL,mythread,&cfd);//mythread为线程执行函数
                //设置线程为分离属性（默认为非分离属性），保证线程执行结束后自行释放所占用资源
                pthread_detach(tid);c++
        }
        close(lfd);
        return 1;

}

注：该代码可以接受数据但无法运行

由于线程创建时与主线程共享内存空间，因此子线程不能关闭lfd（监听描述符），同样主线程也不能关闭cfd（通信描述符）
